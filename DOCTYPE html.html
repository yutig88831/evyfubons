    <script>

        function formatNumber(num) {

            return new Intl.NumberFormat('zh-TW').format(Math.round(num));

        }

 

        function calculate() {

            // Inputs

            const mAmount = parseFloat(document.getElementById('mortgageAmount').value) * 10000;

            const mRate = parseFloat(document.getElementById('mortgageRate').value) / 100;

            const mYears = parseFloat(document.getElementById('mortgageYears').value);

           

            const pAmount = parseFloat(document.getElementById('policyAmount').value) * 10000;

            const pRate = parseFloat(document.getElementById('policyRate').value) / 100;

           

            const iRate = parseFloat(document.getElementById('invRate').value) / 100;

 

            // Calculations

            // Mortgage PMT: P * r * (1+r)^n / ((1+r)^n - 1)

            const monthlyMRate = mRate / 12;

            const mMonths = mYears * 12;

            const monthlyMort = mAmount * (monthlyMRate * Math.pow(1 + monthlyMRate, mMonths)) / (Math.pow(1 + monthlyMRate, mMonths) - 1);

           

            const monthlyPolicy = (pAmount * pRate) / 12;

           

            const totalCapital = mAmount + pAmount;

            const monthlyInv = (totalCapital * iRate) / 12;

           

            const monthlyNet = monthlyInv - monthlyMort - monthlyPolicy;

            const avgLoanRate = ((mAmount * mRate) + (pAmount * pRate)) / totalCapital * 100;

 

            // Update UI

            document.getElementById('totalCapital').innerText = formatNumber(totalCapital);

            document.getElementById('avgLoanRate').innerText = avgLoanRate.toFixed(2);

           

            document.getElementById('monthlyInv').innerText = "+ " + formatNumber(monthlyInv);

            document.getElementById('yearlyInv').innerText = formatNumber(monthlyInv * 12);

           

            document.getElementById('monthlyMort').innerText = "- " + formatNumber(monthlyMort);

            document.getElementById('yearlyMort').innerText = formatNumber(monthlyMort * 12);

            document.getElementById('mortDesc').innerText = `(${mYears}年本息攤還)`;

           

            document.getElementById('monthlyPolicy').innerText = "- " + formatNumber(monthlyPolicy);

            document.getElementById('yearlyPolicy').innerText = formatNumber(monthlyPolicy * 12);

           

            document.getElementById('monthlyNet').innerText = "+ " + formatNumber(monthlyNet);

            document.getElementById('yearlyNet').innerText = formatNumber(monthlyNet * 12);

           

            document.getElementById('marginRate').innerText = (iRate * 100 - avgLoanRate).toFixed(2);

            document.getElementById('bufferValue').innerText = formatNumber(monthlyNet);

           

            // Highlight color if negative (just in case)

            const netElMonthly = document.getElementById('monthlyNet');

            const netElYearly = document.getElementById('yearlyNet');

            if(monthlyNet < 0) {

                netElMonthly.classList.replace('status-positive', 'text-rose-500');

                netElYearly.classList.replace('status-positive', 'text-rose-500');

            } else {

                netElMonthly.classList.add('status-positive');

                netElYearly.classList.add('status-positive');

            }

        }

 

        // Initial Calc

        window.onload = calculate;

    </script>
